src/app/core/guards/auth.guard.ts

authGuard
Type : CanActivateFn
Default value : ( route, state ): Observable<boolean | UrlTree> => { const authService = inject(AuthService); const router = inject(Router); return authService.currentUser.pipe( take(1), map(user => { // 1) Si NO hay usuario → redirigir a /auth if (!user) { return router.createUrlTree(['/auth']); } // 2) Verificar si el perfil está COMPLETO const hasUserInfo = !!user.has_user_information; const hasEducational = !!user.has_educational_user; const hasCategoryInterest = !!user.has_user_category_interest; const needsPersonalize = !hasUserInfo || !hasEducational || !hasCategoryInterest; const targetUrl = state.url || ''; // 3) Si necesita personalizar y NO está ya en /personalize → redirigir if (needsPersonalize && !targetUrl.startsWith('/personalize')) { return router.createUrlTree(['/personalize']); } // 4) Si todo bien → permitir acceso return true; }) ); }

src/app/core/api/auth/auth.interceptor.ts

authInterceptor
Type : HttpInterceptorFn
Default value : (req, next) => { const authService = inject(AuthService); const router = inject(Router); const token = authService.getToken(); // Si no hay token (o está expirado), seguimos normal if (!token) { return next(req); } const authReq = req.clone({ setHeaders: { Authorization: `Bearer ${token}`, Accept: 'application/json' } }); return next(authReq).pipe( catchError((error: HttpErrorResponse) => { if (error.status === 401) { // Token no válido / expirado / revocado authService.clearUserData(); // Redirigir al login (ajusta la ruta si usas otra) router.navigate(['/auth'], { queryParams: { reason: 'session_expired' }, }); } return throwError(() => error); }) ); }

studioGuard
Type : CanActivateFn
Default value : (route, state) => { const authService = inject(AuthService); const router = inject(Router); const user = authService.getCurrentUser(); const isTutor = user?.roles?.some(role => role.name === 'tutor'); const isAdmin = user?.roles?.some(role => role.name === 'admin'); if (authService.isAuthenticated() && (isTutor || isAdmin)) { return true; } // Redirigir al landing si no tiene permiso router.navigate(['**']); return false; }
